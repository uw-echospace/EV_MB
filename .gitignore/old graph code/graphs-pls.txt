import pandas as pd
import matplotlib.pyplot as plt

# Read the CSV file into a DataFrame
data = pd.read_csv('2023-uwa.csv')

# Display the column names to verify their exact names and any leading/trailing whitespace
print(data.columns)

# Check the first few rows of the DataFrame to inspect the data
print(data.head())

# Extract the columns you want to use for plotting
print(data.columns)
print(data.head())
x = data['Date']
y = data['Temperature']

# Create a line plot
plt.plot(x, y)
plt.xlabel('Date')
plt.ylabel('Temperature')
plt.title('Temperature Variation')
plt.show()

# Create a scatter plot
plt.scatter(x, y)
plt.xlabel('X-axis')
plt.ylabel('Y-axis')
plt.title('Scatter Plot')
plt.show()

# Create a bar plot
plt.bar(x, y)
plt.xlabel('X-axis')
plt.ylabel('Y-axis')
plt.title('Bar Plot')
plt.show()
