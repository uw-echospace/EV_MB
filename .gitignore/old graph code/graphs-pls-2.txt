import pandas as pd
import matplotlib.pyplot as plt

# Read the CSV file into a DataFrame, skipping the header row
data = pd.read_csv('2023-uwa.csv', skiprows=1)

# Display the column names to verify their exact names and any leading/trailing whitespace
print(data.columns)

# Check the first few rows of the DataFrame to inspect the data
print(data.head())

# Extract the columns you want to use for plotting
x = data['Date']
y = data['Temperature']

# Create a line plot
plt.plot(x, y)
plt.xlabel('Date')
plt.ylabel('Temperature')
plt.title('Temperature Variation')
plt.show()

# Create a scatter plot
plt.scatter(x, y)
plt.xlabel('Date')
plt.ylabel('Temperature')
plt.title('Temperature Scatter Plot')
plt.show()

# Create a bar plot
plt.bar(x, y)
plt.xlabel('Date')
plt.ylabel('Temperature')
plt.title('Temperature Bar Plot')
plt.show()
